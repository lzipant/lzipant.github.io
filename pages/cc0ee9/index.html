public BinTreeNode<T> generateFromLayerAndInOrder(List<T> layer, List<T> in, int a, int b) {
    if (a > b) {
        return null;
    }
    int index = a;
    while (index <= b && !Objects.equals(layer.get(0), in.get(index))) {
        ++index;

    List<T> leftLayer = new ArrayList<>(index - a);
    List<T> rightLayer = new ArrayList<>(b - index);
    boolean isLeft;
    for (int i = 1; i < layer.size(); ++i) {
        isLeft = false;
        for(int j = a; j < index; ++j) {
            if (Objects.equals(layer.get(i), in.get(j))) {
                isLeft = true;
                break;

        }
        if (isLeft) {
            leftLayer.add(layer.get(i));
        } else {
            rightLayer.add(layer.get(i));
        }
    }
    BinTreeNode<T> node = new BinTreeNode<>();
    node.setData(in.get(index));
    node.setLeft(generateFromLayerAndInOrder(leftLayer, in, a, index - 1));
    node.setRight(generateFromLayerAndInOrder(rightLayer, in, index + 1, b));
    return node;
}
